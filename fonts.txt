  function convertText() {
    const input = document.getElementById("inputText").value;
    let output = "";

    if (currentStyle === 'emoji') {
      for (let char of input) {
        const lower = char.toLowerCase();
        output += styles.emoji.map[lower] || char;
      }
    } else if (currentStyle === 'cursive') {
      for (let char of input) {
        if (/[a-z]/.test(char)) {
          output += String.fromCodePoint(styles.cursive.base.a + (char.charCodeAt(0) - 97));
        } else if (/[A-Z]/.test(char)) {
          output += String.fromCodePoint(styles.cursive.base.A + (char.charCodeAt(0) - 65));
        } else {
          output += char;
        }
      }
    }

    document.getElementById("outputText").value = output;
  }

  function copyText() {
    const text = document.getElementById("outputText");
    text.select();
    document.execCommand("copy");
  }

  function downloadText() {
    const text = document.getElementById("outputText").value;
    const blob = new Blob([text], { type: "text/plain" });
    const link = document.createElement("a");
    link.href = URL.createObjectURL(blob);
    link.download = "converted-text.txt";
    link.click();
  }

  const toolsList = window.TOOL_LIST;

  function showSearchResults(inputValue, container) {
    const query = inputValue.trim().toLowerCase();
    container.innerHTML = '';

    if (query.length < 1) return;

    const results = toolsList.filter(tool => tool.name.toLowerCase().includes(query));

    if (results.length === 0) {
      const noResult = document.createElement('div');
      noResult.textContent = 'No tools found';
      noResult.style.padding = '10px';
      container.appendChild(noResult);
      return;
    }

    results.forEach(tool => {
      const link = document.createElement('a');
      link.href = tool.url;
      link.textContent = tool.name;
      link.style.display = 'block';
      link.style.padding = '10px';
      link.style.textDecoration = 'none';
      link.style.color = '#000';
      link.style.borderBottom = '1px solid #eee';
      link.onmouseover = () => link.style.background = '#f0f0f0';
      link.onmouseout = () => link.style.background = 'transparent';
      container.appendChild(link);
    });
  }

  headerSearchInput.addEventListener('input', () => {
    showSearchResults(headerSearchInput.value, headerSearchResults);
  });

  mobileSearchInput.addEventListener('input', () => {
    showSearchResults(mobileSearchInput.value, mobileSearchResults);
  });

  // Hide results when clicking outside

  }

  // Mobile search results hide

  }
